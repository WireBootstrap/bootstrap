WireBootstrap Table Plugin

Use the WireBootstrap Table plugin to work with data inside a Bootstrap table.

The example below binds an eb.data.DataTable object to the table plugin. 

                    
    var data = [ 
        {product: "apples", order: "PRN001", quantity: 10, price: 10.44},
        {product: "oranges", order: "PRN002", quantity: 15, price: 12.37},
        {product: "apples", order: "PRN003", quantity: 20, price: 13.01}
    ];
    
    var table = new eb.data.DataTable(data);
    
    $("#mytable").ebtable({
        data: table
    });
